*파라미터(parameter)와 아규먼트(argument)

                 |____|------------ 아규먼트를 받는 변수 = 파라미터
function plus(a, b){                        ||
    return a+b;
}		        실무에서는 섞어 쓴다

			||
var result = plus            (100, 200);

		넘겨주는 값 = 아규먼트
